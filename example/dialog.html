属性参数说明(点击显示隐藏)：

dlgClass : "uiBlueBigBorder uiCornerAll uiBlueBoxShadow",//添加到dialog控件的类

titleBarClass : "ui40BlueGradientBG",//标题栏的背景类

position : null,//对话框位置，默认是null,可以设置为"center", "leftTop", 或者[Int, Int]

zIndex : 1000,//dialog控件的z-index,若position为null，该值无效

draggable : false,//是否可以拖动

resizable : false,//是否可以拉伸大小

modal : false,//是否modal模式

buttonSet : null,//dialog的按钮

minHeight : 300,//最小高度

maxHeight : 700,//最大高度

minWidth : 400,//最小宽度

maxWidth : 1000,//最大宽度

active : true,//是否为active

minmizable : true,//是否显示最小化图标

closable : true,//是否显示关闭图标

noTitle : false,//是否无标题栏

titleIcon : null,//标题栏图标

controlButton : true,//是不有标题栏上的控件按钮（最小化，关闭等）

autoOpen : true,//创建完成是否自动打开dialog

destroyOnClose : true,//关闭时是否销毁dialog

closeAnimate : "slideUp",//关闭时调用的动画函数

openAnimate : "slideDown",//打开时调用的动画函数

beforeClose : null,//关闭前调用

close : null,//关闭后调用

beforeOpen : null,//打开前调用

open : null,//打开后调用

beforeMin : null,//最小化前调用

min : null,//最小化后调用

beforeResume : null,//还原前调用

resume : null,//还原后调用

dragStart : null,//拖动前调用

draging : null,//拖动时调用

dragStop : null,//拖动结束时调用

resizeStart : null,//拉缩时调用

resizing : null,//拉伸时调用

resizeStop : null//拉伸结束时调用
