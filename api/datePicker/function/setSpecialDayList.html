var specialDayList = {
                "03/04/2011" : {
                    specialClass : "testTD",
                    desc : "今天是特别的日子"
                },
                "03/21/2011" : {
                    specialClass : "testTD",
                    desc : "今天我没有吃饭"

});

{
    $("#demo").datePicker("setSpecialDayList", specialDayList);
