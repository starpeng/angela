attributedraggable.event

Version Added:0.6.0

Type:Object

Default Value:{start : null, doing : null, stop : null}

Description:拖动时的各状态时调用的方法

HTML

<div id="demo" class="uiNoSelectText uiBlueBigBorder uiCornerAll" style="width:200px;">
                <div class="title ui30BlueGradientBG uiDraggalbe" style="height:30px;line-height:30px;color:white;">拖动测试</div>
                <p>Web.Go 跟 web.py 类似，但使用的是 Go 编程语言实现的 Web 应用开发框架。</p>
            </div>
            <div id="dest" style="width:300px;height:300px;margin:20px;border:1px solid black;"></div>

CODE

$("#demo").draggable({
    event : {
        start : function()
        {
            $("#dest").html("start");
        },
        doing : function(offset)
        {
            $("#dest").html("dragging, left:" + offset.left + ";top:" + offset.top);
        },
        stop : function(offset)
        {
            $("#dest").html("finish left:" + offset.left + ";top:" + offset.top);
        }
    }
});
